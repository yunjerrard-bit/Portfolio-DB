TickerSpec:
    @dataclass(frozen=True)
    class TickerSpec:
        symbol: str
        tier: str = ""        # empty string when 1-column input
        industry: str = ""

TickerResult / TickerFailure:
    @dataclass
    class TickerResult:
        spec: TickerSpec
        enriched_df: pandas.DataFrame   # post-compute, ready for sheet writers
        market: str                     # "US" | "KR"

    @dataclass
    class TickerFailure:
        spec: TickerSpec
        reason: str                     # Korean; e.g., "부분 데이터: 1000 거래일 (예상 2500의 40%)"
                                        # or "yfinance 빈 응답", "네트워크 오류: ConnectionError",
                                        # "잘못된 티커 형식"

run_all signature:
    def run_all(
        specs: list[TickerSpec],
        classify_market: Callable[[str], str],
        pipeline: Callable[[str], pandas.DataFrame],   # fetch+compute for one ticker
    ) -> tuple[list[TickerResult], list[TickerFailure]]

read_tickers_extended:
    def read_tickers_extended(path: str | Path) -> list[TickerSpec]
read_tickers (back-compat):
    def read_tickers(path: str | Path) -> list[str]   # = [s.symbol for s in read_tickers_extended(path)]
